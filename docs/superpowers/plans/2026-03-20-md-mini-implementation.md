# md-mini Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a minimalist live-preview markdown editor for macOS using Tauri 2 + Svelte 5 + CodeMirror 6.

**Architecture:** Tauri 2 provides the native shell (window management, file I/O, CLI, menus). Svelte 5 is a thin wrapper around the CodeMirror 6 editor, which does all the heavy lifting. Live-preview is implemented as a CM6 ViewPlugin that builds Decoration sets from the `@lezer/markdown` syntax tree, skipping decorations when the cursor is within a block.

**Tech Stack:** Tauri 2 (Rust), Svelte 5 (TypeScript), CodeMirror 6, Vite, `@lezer/markdown`, `@floating-ui/dom`

**Spec:** `docs/superpowers/specs/2026-03-19-md-mini-design.md`

```typescript
import { keymap } from '@codemirror/view';
import { defaultKeymap, history, historyKeymap } from '@codemirror/commands';
import { markdown, markdownLanguage } from '@codemirror/lang-markdown';
import { languages } from '@codemirror/language-data';
import { closeBrackets, closeBracketsKeymap } from '@codemirror/autocomplete';
import { searchKeymap } from '@codemirror/search';
import { EditorState, type Extension } from '@codemirror/state';
import { EditorView } from '@codemirror/view';
import { Strikethrough, Table } from '@lezer/markdown';

export function createExtensions(): Extension[] {
  return [
    history(),
    closeBrackets(),
    markdown({
      base: markdownLanguage,
      codeLanguages: languages,
      extensions: [Strikethrough, Table],  // GFM: strikethrough + tables
    }),
    keymap.of([
      ...defaultKeymap,
      ...historyKeymap,
      ...closeBracketsKeymap,
      ...searchKeymap,
    ]),
    EditorView.lineWrapping,
    EditorState.allowMultipleSelections.of(false),
  ];
}
```

```svelte
<script lang="ts">
  import { onMount } from 'svelte';
  import { EditorView } from '@codemirror/view';
  import { EditorState } from '@codemirror/state';
  import { createExtensions } from './setup';

  let { content = '', onchange }: {
    content?: string;
    onchange?: (doc: string) => void;
  } = $props();

  let editorContainer: HTMLDivElement;
  let view: EditorView | undefined = $state(undefined);

  onMount(() => {
    const state = EditorState.create({
      doc: content,
      extensions: [
        ...createExtensions(),
        EditorView.updateListener.of((update) => {
          if (update.docChanged && onchange) {
            onchange(update.state.doc.toString());
          }
        }),
      ],
    });

    view = new EditorView({
      state,
      parent: editorContainer,
    });

    view.focus();

    return () => {
      view?.destroy();
    };
  });

  // To access the view from parent, use bind:this on the component
  // and access view directly — Svelte 5 does not support `export function` in runes mode.
  // Parent can access via: editorComponent.view
</script>
```

- [ ] **Step 6: Create settings store with theme management**

Create `src/lib/stores.ts`:

```typescript
type ThemeSetting = 'light' | 'dark' | 'system';
type EditorMode = 'live-preview' | 'raw';

function loadSetting<T>(key: string, fallback: T): T {
  try {
    const raw = localStorage.getItem(`md-mini:${key}`);
    return raw !== null ? JSON.parse(raw) : fallback;
  } catch {
    return fallback;
  }
}

function saveSetting(key: string, value: unknown): void {
  localStorage.setItem(`md-mini:${key}`, JSON.stringify(value));
}

// --- Theme ---
export function createThemeStore() {
  let preference = $state<ThemeSetting>(loadSetting('theme', 'system'));
  let systemDark = $state(window.matchMedia('(prefers-color-scheme: dark)').matches);

  const resolved = $derived<'light' | 'dark'>(
    preference === 'system' ? (systemDark ? 'dark' : 'light') : preference
  );

  // Listen for system theme changes
  window.matchMedia('(prefers-color-scheme: dark)').addEventListener('change', (e) => {
    systemDark = e.matches;
  });

  return {
    get preference() { return preference; },
    set preference(v: ThemeSetting) {
      preference = v;
      saveSetting('theme', v);
    },
    get resolved() { return resolved; },
  };
}

// --- Editor Mode ---
export function createModeStore() {
  let mode = $state<EditorMode>(loadSetting('mode', 'live-preview'));

  return {
    get value() { return mode; },
    toggle() {
      mode = mode === 'live-preview' ? 'raw' : 'live-preview';
      saveSetting('mode', mode);
    },
  };
}

// --- Zoom ---
export function createZoomStore() {
  let level = $state<number>(loadSetting('zoomLevel', 1.0));

  return {
    get level() { return level; },
    zoomIn() {
      if (level < 2.0) {
        level = Math.round((level + 0.1) * 10) / 10;
        saveSetting('zoomLevel', level);
      }
    },
    zoomOut() {
      if (level > 0.8) {
        level = Math.round((level - 0.1) * 10) / 10;
        saveSetting('zoomLevel', level);
      }
    },
    reset() {
      level = 1.0;
      saveSetting('zoomLevel', level);
    },
  };
}

// --- File State (per window, not persisted) ---
export function createFileState() {
  let filePath = $state<string | null>(null);
  let isDirty = $state(false);
  let lastSavedAt = $state<number | null>(null);

  return {
    get filePath() { return filePath; },
    set filePath(v: string | null) { filePath = v; },
    get isDirty() { return isDirty; },
    set isDirty(v: boolean) { isDirty = v; },
    get lastSavedAt() { return lastSavedAt; },
    set lastSavedAt(v: number | null) { lastSavedAt = v; },
    get title() {
      const name = filePath ? filePath.split('/').pop() : 'Untitled';
      return `${isDirty ? '● ' : ''}${name} — md-mini`;
    },
  };
}

// --- Recent Files ---
interface RecentFile {
  path: string;
  timestamp: number;
}

export function createRecentFilesStore() {
  let files = $state<RecentFile[]>(loadSetting('recentFiles', []));

  return {
    get list() { return files; },
    add(path: string) {
      files = [
        { path, timestamp: Date.now() },
        ...files.filter((f) => f.path !== path),
      ].slice(0, 10);
      saveSetting('recentFiles', files);
    },
  };
}
```

```rust
        format!("Failed to save: {}", e)
    })
}
```

```typescript
function toggleWrap(view: EditorView, marker: string): boolean {
  const { state } = view;
  const changes = state.changeByRange((range) => {
    const text = state.sliceDoc(range.from, range.to);
    const len = marker.length;

    // Check if already wrapped
    if (text.startsWith(marker) && text.endsWith(marker) && text.length >= len * 2) {
      // Remove markers
      return {
        changes: [{ from: range.from, to: range.to, insert: text.slice(len, -len) }],
        range: EditorSelection.range(range.from, range.to - len * 2),
      };
    }

    // Check surrounding context
    const before = state.sliceDoc(Math.max(0, range.from - len), range.from);
    const after = state.sliceDoc(range.to, Math.min(state.doc.length, range.to + len));

    if (before === marker && after === marker) {
      // Remove surrounding markers
      return {
        changes: [
          { from: range.from - len, to: range.from, insert: '' },
          { from: range.to, to: range.to + len, insert: '' },
        ],
        range: EditorSelection.range(range.from - len, range.to - len),
      };
    }

    // Add markers
    return {
      changes: [{ from: range.from, to: range.to, insert: `${marker}${text}${marker}` }],
      range: EditorSelection.range(range.from + len, range.to + len),
    };
  });

  view.dispatch(changes);
  return true;
}

export function markdownKeybindings(): Extension {
  return keymap.of([
    {
      key: 'Mod-b',
      run: (view) => toggleWrap(view, '**'),
    },
    {
      key: 'Mod-i',
      run: (view) => toggleWrap(view, '*'),
    },
    {
      key: 'Mod-Shift-x',
      run: (view) => toggleWrap(view, '~~'),
    },
  ]);
}
```

```typescript
import { keymap } from '@codemirror/view';
import type { Extension } from '@codemirror/state';
import type { EditorView } from '@codemirror/view';

/**
 * List continuation on Enter:
 * - `- item|` → new line with `- `
 * - `* item|` → new line with `* `
 * - `1. item|` → new line with `2. `
 * - `- [ ] item|` → new line with `- [ ] `
 * - Empty list item → exit list (remove marker)
 */
function handleEnterInList(view: EditorView): boolean {
  const { state } = view;
  const { from } = state.selection.main;
  const line = state.doc.lineAt(from);
  const text = line.text;

  // Check for list patterns
  const bulletMatch = text.match(/^(\s*)([-*+])\s(.*)$/);
  const numberedMatch = text.match(/^(\s*)(\d+)\.\s(.*)$/);
  const checkboxMatch = text.match(/^(\s*)([-*+])\s\[[ x]\]\s(.*)$/);

  if (checkboxMatch) {
    const [, indent, marker, content] = checkboxMatch;
    if (content.trim() === '') {
      // Empty checkbox item → remove it
      view.dispatch({ changes: { from: line.from, to: line.to, insert: '' } });
      return true;
    }
    const continuation = `\n${indent}${marker} [ ] `;
    view.dispatch({
      changes: { from, insert: continuation },
      selection: { anchor: from + continuation.length },
    });
    return true;
  }

  if (bulletMatch) {
    const [, indent, marker, content] = bulletMatch;
    if (content.trim() === '') {
      view.dispatch({ changes: { from: line.from, to: line.to, insert: '' } });
      return true;
    }
    const continuation = `\n${indent}${marker} `;
    view.dispatch({
      changes: { from, insert: continuation },
      selection: { anchor: from + continuation.length },
    });
    return true;
  }

  if (numberedMatch) {
    const [, indent, num, content] = numberedMatch;
    if (content.trim() === '') {
      view.dispatch({ changes: { from: line.from, to: line.to, insert: '' } });
      return true;
    }
    const next = parseInt(num, 10) + 1;
    const continuation = `\n${indent}${next}. `;
    view.dispatch({
      changes: { from, insert: continuation },
      selection: { anchor: from + continuation.length },
    });
    return true;
  }

  // Code fence auto-close: ``` + Enter → insert closing ```
  const fenceMatch = text.match(/^(\s*)(`{3,})(\w*)\s*$/);
  if (fenceMatch && from === line.to) {
    const [, indent, ticks] = fenceMatch;
    const insert = `\n${indent}\n${indent}${ticks}`;
    view.dispatch({
      changes: { from, insert },
      selection: { anchor: from + 1 + indent.length }, // cursor on empty line inside
    });
    return true;
  }

  return false; // Let default Enter handle it
}

/**
 * Tab/Shift-Tab for list indent/outdent
 */
function handleTabInList(view: EditorView, indent: boolean): boolean {
  const { state } = view;
  const { from } = state.selection.main;
  const line = state.doc.lineAt(from);
  const text = line.text;

  // Check if we're in a list
  if (!/^\s*[-*+\d]/.test(text)) return false;

  if (indent) {
    view.dispatch({
      changes: { from: line.from, insert: '  ' },
      selection: { anchor: from + 2 },
    });
  } else {
    if (text.startsWith('  ')) {
      view.dispatch({
        changes: { from: line.from, to: line.from + 2, insert: '' },
        selection: { anchor: Math.max(line.from, from - 2) },
      });
    }
  }
  return true;
}

export function listContinuation(): Extension {
  return keymap.of([
    { key: 'Enter', run: handleEnterInList },
    { key: 'Tab', run: (view) => handleTabInList(view, true) },
    { key: 'Shift-Tab', run: (view) => handleTabInList(view, false) },
  ]);
}
```

```typescript
  builder: RangeSetBuilder<Decoration>
```

```typescript
class CheckboxWidget extends WidgetType {
  constructor(private checked: boolean, private pos: number) {
    super();
  }

  toDOM(view: EditorView): HTMLElement {

    return false;

  toDOM(): HTMLElement {
    const span = document.createElement('span');

    span.textContent = '\u2022'; // bullet dot
    return span;

    return true;

export function decorateListItem(
  view: EditorView,
  node: SyntaxNode,
  builder: RangeSetBuilder<Decoration>
): void {
  if (cursorInRange(view, node.from, node.to)) return;

  const listMark = node.getChild('ListMark');
  if (!listMark) return;

  const doc = view.state.doc;
  const afterMark = doc.sliceDoc(listMark.to, Math.min(listMark.to + 5, doc.length));

  // Check for checkbox pattern: [ ] or [x]
  const checkboxMatch = afterMark.match(/^\s\[([x ])\]/);
  if (checkboxMatch) {
    const isChecked = checkboxMatch[1] === 'x';
    const checkboxStart = listMark.to + 1; // space after marker
    // Replace "- [ ] " or "- [x] " with checkbox widget
    builder.add(
      listMark.from,
      checkboxStart + 3,
      Decoration.replace({
        widget: new CheckboxWidget(isChecked, checkboxStart),
      })
    );
    return;
  }

  // Bullet list: replace marker with styled bullet
  const markText = doc.sliceDoc(listMark.from, listMark.to);
  if (markText === '-' || markText === '*' || markText === '+') {
    builder.add(
      listMark.from,
      listMark.to,
      Decoration.replace({ widget: new BulletWidget() })
    );
  }
  // Numbered lists: keep number, just style the line
}

export function decorateBlockquote(
  view: EditorView,
  node: SyntaxNode,
  builder: RangeSetBuilder<Decoration>
): void {
  if (cursorInRange(view, node.from, node.to, true)) return;

  // Add line decoration for blockquote styling
  const doc = view.state.doc;
  const startLine = doc.lineAt(node.from);
  const endLine = doc.lineAt(node.to);

  for (let i = startLine.number; i <= endLine.number; i++) {
    const line = doc.line(i);
    builder.add(line.from, line.from, Decoration.line({ class: 'cm-md-blockquote' }));

    // Hide the `> ` marker
    const text = line.text;
    const match = text.match(/^(\s*>)\s?/);
    if (match) {
      builder.add(line.from, line.from + match[0].length, Decoration.replace({}));
    }
  }
}
```

```typescript
/**
 * Code blocks: uses line decorations (hide fences, style code lines).
 * NOTE: The spec calls for a WidgetType with syntax highlighting, but that
 * introduces cursor navigation complexity (atomic widgets). This simpler approach
 * hides fence lines and styles code lines — CM6's built-in markdown syntax
 * highlighting still applies to the code content. The widget approach can be
 * implemented as a v1.1 enhancement if visual quality is insufficient.
 */
export function decorateFencedCode(
  view: EditorView,
  node: SyntaxNode,
  builder: RangeSetBuilder<Decoration>
): void {
  if (cursorInRange(view, node.from, node.to, true)) return;

  // Style the code block without replacing it — just add line decorations
  const doc = view.state.doc;
  const startLine = doc.lineAt(node.from);
  const endLine = doc.lineAt(node.to);

  // Hide opening fence line (replace within line only, don't cross boundaries)
  builder.add(startLine.from, startLine.from, Decoration.line({ class: 'cm-md-code-fence-hidden' }));

  // Style code lines
  for (let i = startLine.number + 1; i < endLine.number; i++) {
    const line = doc.line(i);
    builder.add(line.from, line.from, Decoration.line({ class: 'cm-md-code-line' }));
  }

  // Hide closing fence line
  if (endLine.number > startLine.number) {
    const closingLine = doc.line(endLine.number);
    builder.add(closingLine.from, closingLine.from, Decoration.line({ class: 'cm-md-code-fence-hidden' }));
  }

  // Add wrapper line decoration on first code line
  if (startLine.number + 1 <= endLine.number - 1) {
    const firstCodeLine = doc.line(startLine.number + 1);
    builder.add(firstCodeLine.from, firstCodeLine.from, Decoration.line({ class: 'cm-md-code-block-start' }));
    if (endLine.number - 1 > startLine.number) {
      const lastCodeLine = doc.line(endLine.number - 1);
      builder.add(lastCodeLine.from, lastCodeLine.from, Decoration.line({ class: 'cm-md-code-block-end' }));
    }
  }
}
```

```typescript
import { Decoration, WidgetType } from '@codemirror/view';

function parseMarkdownTable(text: string): { headers: string[]; rows: string[][] } {
  const lines = text.split('\n').filter((l) => l.trim());
  if (lines.length < 2) return { headers: [], rows: [] };

  const parseRow = (line: string): string[] =>
    line.split('|').map((c) => c.trim()).filter((_, i, arr) => i > 0 && i < arr.length);

  const headers = parseRow(lines[0]);
  // Skip separator line (index 1)
  const rows = lines.slice(2).map(parseRow);

  return { headers, rows };
}

class TableWidget extends WidgetType {
  constructor(private tableText: string) {
    super();
  }

  toDOM(): HTMLElement {
    const { headers, rows } = parseMarkdownTable(this.tableText);

    const wrapper = document.createElement('div');
    wrapper.className = 'cm-md-table-wrapper';

    const table = document.createElement('table');
    table.className = 'cm-md-table';

    // Header
    if (headers.length > 0) {
      const thead = document.createElement('thead');
      const tr = document.createElement('tr');
      for (const h of headers) {
        const th = document.createElement('th');
        th.textContent = h;
        tr.appendChild(th);
      }
      thead.appendChild(tr);
      table.appendChild(thead);
    }

    // Body
    if (rows.length > 0) {
      const tbody = document.createElement('tbody');
      for (const row of rows) {
        const tr = document.createElement('tr');
        for (const cell of row) {
          const td = document.createElement('td');
          td.textContent = cell;
          tr.appendChild(td);
        }
        tbody.appendChild(tr);
      }
      table.appendChild(tbody);
    }

    wrapper.appendChild(table);
    return wrapper;
  }

  eq(other: TableWidget): boolean {
    return this.tableText === other.tableText;
  }
}

  builder.add(
    node.from,
```

```typescript
import type { CompletionContext, CompletionResult } from '@codemirror/autocomplete';
import { EditorState, type Extension } from '@codemirror/state';

interface SlashCommand {
  label: string;
  detail: string;
  insert: string;
  cursorOffset?: number; // offset from end of insert to place cursor
}

const commands: SlashCommand[] = [
  { label: '/heading1', detail: 'Heading 1', insert: '# ' },
  { label: '/heading2', detail: 'Heading 2', insert: '## ' },
  { label: '/heading3', detail: 'Heading 3', insert: '### ' },
  { label: '/heading4', detail: 'Heading 4', insert: '#### ' },
  { label: '/heading5', detail: 'Heading 5', insert: '##### ' },
  { label: '/heading6', detail: 'Heading 6', insert: '###### ' },
  { label: '/bullet', detail: 'Bullet list', insert: '- ' },
  { label: '/numbered', detail: 'Numbered list', insert: '1. ' },
  { label: '/checkbox', detail: 'Checkbox', insert: '- [ ] ' },
  { label: '/code', detail: 'Code block', insert: '```\n\n```', cursorOffset: -4 },
  { label: '/table', detail: 'Table', insert: '| Column 1 | Column 2 |\n|----------|----------|\n|          |          |\n' },
  { label: '/quote', detail: 'Blockquote', insert: '> ' },
  { label: '/hr', detail: 'Horizontal rule', insert: '---\n' },
];

function slashCommandSource(context: CompletionContext): CompletionResult | null {
  // Match "/" at start of line or after list indentation
  const before = context.matchBefore(/(?:^|\n)\s*\/\w*/);
  if (!before) return null;

  const slashIndex = before.text.lastIndexOf('/');
  const from = before.from + slashIndex;

  return {
    from,
    options: commands.map((cmd) => ({
      label: cmd.label,
      detail: cmd.detail,
      apply: (view, _completion, applyFrom, applyTo) => {
        const insert = cmd.insert;
        view.dispatch({
          changes: { from: applyFrom, to: applyTo, insert },
          selection: cmd.cursorOffset
            ? { anchor: applyFrom + insert.length + cmd.cursorOffset }
            : { anchor: applyFrom + insert.length },
        });
      },
    })),
  };
}

/**
 * Returns the slash command CompletionSource.
 * Do NOT use autocompletion({ override: [...] }) as it replaces all sources
 * including closeBrackets. Instead, add this as an additional source
 * via the languageData completion facet or a separate autocompletion() call
 * that doesn't use `override`.
 */
export function slashCommands(): Extension {
  return EditorState.languageData.of(() => [
    { autocomplete: slashCommandSource },
  ]);
}
```

```typescript
let activePopup: HTMLElement | null = null;

function hidePopup() {
  if (activePopup) {
    activePopup.remove();
    activePopup = null;
  }
}

function showPopup(view: EditorView, button: HTMLElement, lineFrom: number) {
  hidePopup();

  const popup = document.createElement('div');
  popup.className = 'cm-block-menu-popup';

  for (const cmd of blockCommands) {
    const item = document.createElement('div');
    item.className = 'cm-block-menu-item';
    item.textContent = cmd.label;
    item.addEventListener('mousedown', (e) => {
      e.preventDefault();
      hidePopup();

      const line = view.state.doc.lineAt(lineFrom);
      const insert = cmd.insert;
      const from = line.from;
      const to = line.to;

      view.dispatch({
        changes: { from, to, insert: insert + (line.text.length > 0 ? '' : '') },
        selection: cmd.cursorOffset
          ? { anchor: from + insert.length + cmd.cursorOffset }
          : { anchor: from + insert.length },
      });
      view.focus();
    });
    popup.appendChild(item);
  }

  document.body.appendChild(popup);
  activePopup = popup;

  computePosition(button, popup, {
    placement: 'right-start',
    middleware: [offset(4), flip()],
  }).then(({ x, y }) => {
    popup.style.left = `${x}px`;
    popup.style.top = `${y}px`;
  });

  // Close on click outside
  const closeHandler = (e: MouseEvent) => {
    if (!popup.contains(e.target as Node) && e.target !== button) {
      hidePopup();
      document.removeEventListener('mousedown', closeHandler);
    }
  };
  setTimeout(() => document.addEventListener('mousedown', closeHandler), 0);
}

    button.textContent = '+';
    button.addEventListener('mousedown', (e) => {
      e.preventDefault();
      e.stopPropagation();
      showPopup(view, button, this.lineFrom);

    return this.lineFrom === other.lineFrom;
```

```rust
use std::sync::{atomic::{AtomicU32, Ordering}, Mutex};
use std::collections::HashMap;
use tauri::{AppHandle, Manager, WebviewWindowBuilder, WebviewUrl, Emitter};

static WINDOW_COUNTER: AtomicU32 = AtomicU32::new(0);
static CASCADE_OFFSET: f64 = 25.0;

/// Tracks which file is open in which window (for deduplication)
pub struct OpenFiles(pub Mutex<HashMap<String, String>>); // path -> window_label

pub fn open_file_window(app: &AppHandle, file_path: Option<String>) -> tauri::Result<()> {
    // Dedup: if file already open, focus that window
    if let Some(ref path) = file_path {
        let open_files = app.state::<OpenFiles>();
        let files = open_files.0.lock().unwrap();
        if let Some(label) = files.get(path) {
            if let Some(window) = app.get_webview_window(label) {
                let _ = window.set_focus();
                return Ok(());
            }
        }
    }

    let count = WINDOW_COUNTER.fetch_add(1, Ordering::SeqCst);
    let label = format!("editor-{}", count);
    let offset = (count as f64) * CASCADE_OFFSET;

    let _window = WebviewWindowBuilder::new(
        app,
        &label,
        WebviewUrl::App("index.html".into()),
    )
    .title(match &file_path {
        Some(p) => format!("{} — md-mini", p.split('/').last().unwrap_or("Untitled")),
        None => "Untitled — md-mini".to_string(),
    })
    .inner_size(900.0, 700.0)
    .min_inner_size(400.0, 300.0)
    .position(100.0 + offset, 100.0 + offset)
    .build()?;

    // Track open file
    if let Some(ref path) = file_path {
        let open_files = app.state::<OpenFiles>();
        open_files.0.lock().unwrap().insert(path.clone(), label.clone());
    }

    // Send file path to the new window after it loads
    if let Some(path) = file_path {
        let handle = app.clone();
        let label_clone = label.clone();
        tauri::async_runtime::spawn(async move {
            tokio::time::sleep(std::time::Duration::from_millis(300)).await;
            if let Some(window) = handle.get_webview_window(&label_clone) {
                let _ = window.emit("open-file", &path);
            }
        });
    }

    Ok(())
}
```

```rust
pub fn watch_file(app: AppHandle, path: String, window_label: String) -> notify::Result<notify::RecommendedWatcher> {
    let (tx, rx) = mpsc::channel::<notify::Result<Event>>();

    let mut watcher = notify::recommended_watcher(tx)?;
    watcher.watch(Path::new(&path), RecursiveMode::NonRecursive)?;

    // Spawn thread to forward events
    std::thread::spawn(move || {
        for result in rx {
            if let Ok(event) = result {
                if matches!(event.kind, EventKind::Modify(_)) {
                    if let Some(window) = app.get_webview_window(&window_label) {
                        let _ = window.emit("file-changed-externally", &path);
                    }
                }
            }
        }
    });

    Ok(watcher)
}
```

```rust
use std::collections::hash_map::DefaultHasher;
use std::hash::{Hash, Hasher};
use std::fs;
use std::path::PathBuf;
use tauri::command;

fn recovery_dir() -> PathBuf {
    let mut dir = dirs::data_dir().unwrap_or_else(|| PathBuf::from("."));
    dir.push("md-mini");
    dir.push("recovery");
    let _ = fs::create_dir_all(&dir);
    dir
}

fn hash_path(path: &str) -> String {
    let mut hasher = DefaultHasher::new();
    path.hash(&mut hasher);
    format!("{:x}", hasher.finish())
}

#[command]
pub async fn save_recovery(path: String, content: String) -> Result<(), String> {
    let dir = recovery_dir();
    let hash = hash_path(&path);
    let recovery_path = dir.join(format!("{}.md", hash));

    let timestamp = std::time::SystemTime::now()
        .duration_since(std::time::UNIX_EPOCH)
        .unwrap_or_default()
        .as_secs();
    let header = format!("<!-- recovery: {} -->\n<!-- timestamp: {} -->\n", path, timestamp);
    fs::write(&recovery_path, format!("{}{}", header, content))
        .map_err(|e| e.to_string())
}

#[command]
pub async fn delete_recovery(path: String) -> Result<(), String> {
    let dir = recovery_dir();
    let hash = hash_path(&path);
    let recovery_path = dir.join(format!("{}.md", hash));
    if recovery_path.exists() {
        fs::remove_file(&recovery_path).map_err(|e| e.to_string())?;
    }
    Ok(())
}

#[command]
pub async fn check_recovery() -> Result<Vec<(String, String)>, String> {
    let dir = recovery_dir();
    if !dir.exists() {
        return Ok(vec![]);
    }

    let mut results = vec![];
    for entry in fs::read_dir(&dir).map_err(|e| e.to_string())? {
        let entry = entry.map_err(|e| e.to_string())?;
        let content = fs::read_to_string(entry.path()).map_err(|e| e.to_string())?;

        // Extract original path from header
        if let Some(line) = content.lines().next() {
            if let Some(path) = line.strip_prefix("<!-- recovery: ").and_then(|s| s.strip_suffix(" -->")) {
                // Return path and content (without header)
                let body = content.lines().skip(2).collect::<Vec<_>>().join("\n");
                results.push((path.to_string(), body));
            }
        }
    }
    Ok(results)
}
```

## Notes for Implementer

1. **Decoration ordering matters.** CM6 `RangeSetBuilder` requires decorations to be added in document order (ascending `from`). The `syntaxTree.iterate()` traverses in order, but within a single node handler, ensure `builder.add()` calls are ordered.

2. **`@lezer/markdown` node names** may vary slightly. Use the CM6 dev tools extension during development to inspect the syntax tree:
   ```typescript
   import { showSyntaxTree } from '@codemirror/language';
   // Add to extensions temporarily for debugging
   ```

3. **Performance:** The ViewPlugin rebuilds decorations on every selection change. For large documents, use `view.visibleRanges` to only decorate visible content.

4. **Tauri 2 APIs** may differ from examples. Always check the latest Tauri 2 stable docs. The `@tauri-apps/api` package must match the Rust `tauri` version.

5. **Font file names** may differ from what's listed. Adjust `@font-face` declarations to match actual filenames after download.

6. **GFM (GitHub Flavored Markdown)** extensions for `@lezer/markdown`: tables and strikethrough require GFM extensions to be enabled:
   ```typescript
   import { Strikethrough, Table } from '@lezer/markdown';
   markdown({ base: markdownLanguage, codeLanguages: languages, extensions: [Strikethrough, Table] })
   ```
